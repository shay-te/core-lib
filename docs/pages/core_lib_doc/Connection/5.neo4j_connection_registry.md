`Neo4jConnection` provides a `neo4j.Session` on every call to the `get` method. The connection is closed when exiting the `"with"` statement.



### Example

### `your_core_lib.yaml`

```yaml
core_lib:
  data:
    neo4j:
      credentials:
        username: ${oc.env:NEO4J_USERNAME}
        password: ${oc.env:NEO4J_PASSWORD}
      url:
        host: neo4j
        port: 7474
```



### `your_core_lib.py`

```python
from core_lib.connection.neo4j_connection_factory import Neo4jConnectionFactory


class YourCoreLib(CoreLib):
    def __init__(self, conf: DictConfig):
        super().__init__()
        neo4j = Neo4jConnectionFactory(conf.core_lib.data.neo4j)
        self.search = SearchDataAccess(neo4j)
```



### `search_data_access.py`

```python
class SearchDataAccess(DataAccess):

    ALLOW_UPDATE_FIELDS = ['name', 'gender']

    def __init__(self, db: Neo4jConnectionFactory):
        self._db = db

    def search(self, name: str):
        with self._db.get() as session:
            cypher = 'MATCH (me:Book {name:$name}) ...'
            result = session.run(cypher, {'name': name})
            return [dict(r) for r in result]

    def insert(self, data: dict):
        parameters = {
            'user_id': data['id'],
            'name': data['nick_name'],
            'gender': data['gender'],
        }
        with self._db.get() as session:
            return session.run('MERGE (u:User {user_id:$user_id}) ON CREATE SET u.name=$name, ...', parameters)

    def update(self, user_id: int, data: dict):
        query_fields = []
        update_data = {}
        for key, value in data.items():
            if key in self.ALLOW_UPDATE_FIELDS:
                query_fields.append(f'u.{key}=${key}')
                update_data[key] = value
        if query_fields:
            fields_str = ','.join(query_fields)
            update_data['user_id'] = user_id
            with self._db.get() as session:
                return session.run(f'MATCH (u:User {{ user_id: $user_id }}) SET {fields_str} RETURN u', update_data)
```
